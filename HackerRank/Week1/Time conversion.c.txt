char* timeConversion(char* s) {
    //string s = "00:00:00";
    
  static char result[9];
  
  int hour;
  char period[3];
  
  sscanf(s,"%2d:%*2d:%*2d%2s", &hour,period);
  
  if(strcmp(period, "AM")== 0){
    if(hour == 12)
    hour = 0;
  } else {
    if(hour != 12)
     hour += 12;
  }
  
  snprintf(result, sizeof(result),"%02d%s",hour, s + 2);
  
  result[8] = '\0';
  
  return result ;
}
