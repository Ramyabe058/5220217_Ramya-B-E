void miniMaxSum(int arr_count, int* arr) {
    long min= arr[0],max= arr[0];
    long sum = 0;
    
    for(int i = 0; i < arr_count; i++){
        
        sum += arr[i];
        
        if(arr[i] < min)
            min = arr[i];
        if(arr[i] > max)
            max = arr[i];       
    }
    
    long min_sum = sum - max;
    long max_sum = sum - min;
    
    printf("%ld %ld\n",min_sum,max_sum);
}
